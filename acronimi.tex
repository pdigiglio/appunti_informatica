\chapter{Elenco degli acronimi}
\begin{acronym}[MADAM]
	\acro{cpu}[CPU]{\foreignlanguage{english}Central Process Unit}
{\small\par\noindent L'\emph{Unità Centrale d'Elaborazione} è la parte del computer che si occupa di trasformare e manipolare i dati.}
	\acro{ram}[RAM]{\foreignlanguage{english}Random Access Memory}
{\small\par\noindent La \emph{Memoria di Lavoro} conserva i dati durante l'esecuzione di un programma o la loro elaborazione.}
	\acro{io}[I/O]{\foreignlanguage{english}Input/Output}
{\small\par\noindent Le unità di \emph{Ingresso/Uscita} consentono di immettere/acquisire i dati dal calcolatore}
	\acro{lifo}[LIFO]{\foreignlanguage{english}Last In First Out}
	\acro{fifo}[FIFO]{\foreignlanguage{english}First In First Out}
	\acro{bst}[BST]{\foreignlanguage{english}Binary Search Tree}
%{\small\par\noindent Le unità di \emph{Ingresso/Uscita} consentono di immettere/acquisire i dati dal calcolatore}
	\acro{ascii}[ASCII]{\foreignlanguage{english}American Standard Code for Information Interchange}
	\acro{os}[OS]{\foreignlanguage{english}Operating System}
	\acro{acm}[ACM]{\foreignlanguage{english}Association for Computing Machinery}
	\acro{npl}[NPL]{\foreignlanguage{english}National Physical Laboratory}
	\acro{ace}[ACE]{\foreignlanguage{english}Automatic Computing Engine}
	\acro{madam}[MADAM]{\foreignlanguage{english}Manchester Automatical Digital Machine}
\end{acronym}
